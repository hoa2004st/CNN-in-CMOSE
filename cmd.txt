apt update

  torch.cuda.is_available()); print('device_count=',
  torch.cuda.device_count()); print('device_name=',
  torch.cuda.get_device_name(0) if torch.cuda.is_available() else
  'N/A')"
  
# Clone repo
git clone https://github.com/hoa2004st/CNN-in-CMOSE
cd CNN-in-CMOSE

# Python environment
python3 -m venv .venv
source .venv/bin/activate
python -m pip install --upgrade pip
pip install -r requirements.txt
pip install "huggingface_hub[cli]"

# Download CMOSE dataset files
mkdir -p data/CMOSE
hf download cwuau/CMOSE --repo-type dataset --local-dir ./data/CMOSE
unzip -o data/CMOSE/secondFeature.zip -d data/CMOSE/secondFeature
rm -f data/CMOSE/secondFeature.zip

# Verify that PyTorch can see the GPU
python -c "import torch; print('cuda_available=', torch.cuda.is_available()); print('device_count=', torch.cuda.device_count()); print('device_name=', torch.cuda.get_device_name(0) if torch.cuda.is_available() else 'N/A')"

# Quick smoke test on one raw-sequence model
python main.py --model temporal_cnn --device cuda --amp --epochs 2 --batch_size 4 --output_dir outputs/smoke_temporal_cnn

# Baseline runs under the strict no-leakage protocol
python main.py --model paper_cnn --method svd --device cuda --amp --output_dir outputs/paper_cnn_svd
python main.py --model paper_cnn --method pca --device cuda --amp --output_dir outputs/paper_cnn_pca

# Single comparison-model runs
python main.py --model temporal_cnn --device cuda --amp --output_dir outputs/temporal_cnn
python main.py --model rectangular_cnn --device cuda --amp --output_dir outputs/rectangular_cnn
python main.py --model spectral_cnn --device cuda --amp --output_dir outputs/spectral_cnn
python main.py --model lstm --device cuda --amp --output_dir outputs/lstm
python main.py --model transformer --device cuda --amp --output_dir outputs/transformer

# Loss-function variants for imbalanced training
python main.py --model temporal_cnn --device cuda --amp --loss weighted_cross_entropy --output_dir outputs/temporal_cnn_weighted_ce
python main.py --model temporal_cnn --device cuda --amp --loss focal --focal_gamma 2.0 --output_dir outputs/temporal_cnn_focal
python main.py --model temporal_cnn --device cuda --amp --loss ordinal --output_dir outputs/temporal_cnn_ordinal
python main.py --model lstm --device cuda --amp --loss weighted_cross_entropy --output_dir outputs/lstm_weighted_ce
python main.py --model lstm --device cuda --amp --loss ordinal --output_dir outputs/lstm_ordinal
python main.py --model transformer --device cuda --amp --loss focal --focal_gamma 2.0 --output_dir outputs/transformer_focal
python main.py --model transformer --device cuda --amp --loss ordinal --output_dir outputs/transformer_ordinal

chmod +x scripts/run_comparison_models.sh
bash scripts/run_comparison_models.sh
